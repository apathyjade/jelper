## 样例

import EnumHelper from '@jelper/enum';

export const sad = 123;

```jsx live noInline
const enumHelper = EnumHelper.build([
  { label: 'A', key: 'a', value: 1 },
  { label: 'A', key: 'a', value: 1 },
  { label: 'A', key: 'a', value: 1 },
], {
  getTotal() {
    return this.valueOf.reduce((res, it) => res + it.value);
  }
})
render(
  <>
    <div> a: {enumHelper.getLabel('a')} </div>
    <div>total: {enumHelper.getTotal()}</div>
  </>
);
```
