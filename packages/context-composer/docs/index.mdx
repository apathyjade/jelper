### 样例

```tsx live noInline

// hooks/useStateCtx.tsx
const useStateHooks = (props) => {
  const [count, setCount] = useState(props.count);
  return [count, setCount];
}
const { forwardComposer, useComposerCtx } = createCtxComposer(useStateHooks);

// pages/father.tsx
const Father = (porps) => {
  const [count, setCount] = useComposerCtx();
  return (
    <div>
      <div>{JSON.stringify(porps)}</div>
      {count}
      <div onClick={() => setCount(count + 1)} > + </div>
    </div>
  )
}
const Demo = forwardComposer(Father);

// 
render(
  <Demo count={6} />
)
```
