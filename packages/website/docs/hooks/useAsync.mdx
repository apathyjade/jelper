import { useState, useRef, useCallback, useEffect, useMemo } from 'react';
import { useCreate, useRtCb, useRtState } from '@jelper/hooks';


export const Demo = () => {
    const [state, setState] = useState(0);
    const onClick = useCallback(() => {
        console.log('useCallback')
        setState((val) => val + 1);
    }, [])
    useEffect(() => {
        console.log('useEffect', state)
        return () => {
            console.log('demo useEffect', state)
        }
    }, [state])
    return <span onClick={onClick} >{state}</span>;
}
