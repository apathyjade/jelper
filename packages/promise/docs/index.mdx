import { useEffect } from 'react';
import { useSingleHandler } from '@jelper/promise';

export const handler = useSingleHandler();

export const Demo = () => {
  useEffect(() => {
    handler.on(() => {
      console.log('on');
      return { state: 0 }
    })
  }, [])
  return <p onClick={() => handler.call().then(console.log) }>测试下</p>;
};
